    <script>
        function switchTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));
            document.getElementById('page-' + tabId).classList.remove('hidden');
            document.querySelectorAll('.tab-btn').forEach(btn => {
                btn.classList.remove('bg-indigo-600', 'text-white', 'shadow-sm');
                btn.classList.add('text-slate-500');
            });
            document.getElementById('btn-' + tabId).classList.add('bg-indigo-600', 'text-white', 'shadow-sm');
            document.getElementById('btn-' + tabId).classList.remove('text-slate-500');
        }

        function startDay1() {
            document.getElementById('day-workspace').classList.remove('hidden');
            document.getElementById('day-workspace').scrollIntoView({ behavior: 'smooth' });
        }

        function checkAnswer(isCorrect, optId) {
            const resultBox = document.getElementById('quiz-result');
            resultBox.classList.remove('hidden');
            if (isCorrect) {
                resultBox.className = "p-4 rounded-xl text-sm border border-emerald-200 bg-emerald-50 text-emerald-700";
                resultBox.innerHTML = "🎯 إجابة استشارية ممتازة! حصلت على +25 نقطة.";
                localStorage.setItem('od_points', '25');
                localStorage.setItem('od_rank', 'محلل تنظيمي');
                localStorage.setItem('badge_diagnostic', 'unlocked');
                updateDashboardUI();
            }
        }

        function updateDashboardUI() {
            document.getElementById('user-points').innerText = (localStorage.getItem('od_points') || '0') + " / 600";
            document.getElementById('user-rank').innerText = localStorage.getItem('od_rank') || 'متدرب OD مستجد';
            if (localStorage.getItem('badge_diagnostic') === 'unlocked') {
                document.getElementById('badge-diagnostic').classList.remove('opacity-20', 'grayscale');
            }
        }
        window.onload = updateDashboardUI;
    </script>
